   142	class MemoryFusion(nn.Module):
   143	    """Fuse retrieved memory with the recurrent state."""
   144	
   145	    def __init__(
   146	        self,
   147	        *,
   148	        hidden_dim: int,
   149	        value_dim: int,
   150	        mode: FusionName,
   151	    ) -> None:
   152	        super().__init__()
   153	        self.mode = mode
   154	        self.memory_projection = nn.Linear(value_dim, hidden_dim)
   155	        self.concatenate = nn.Linear(hidden_dim * 2, hidden_dim)
   156	        self.gate = nn.Linear(hidden_dim * 2, hidden_dim)
   157	        self.attention_score = nn.Linear(hidden_dim, 1)
   158	        self.normalization = nn.LayerNorm(hidden_dim)
   159	
   160	    def forward(self, hidden: Tensor, retrieved: Tensor) -> Tensor:
   161	        projected = self.memory_projection(retrieved)
   162	        if self.mode == "concatenation":
   163	            return torch.tanh(self.concatenate(torch.cat([hidden, projected], dim=-1)))
   164	        if self.mode == "residual":
   165	            return self.normalization(hidden + projected)
   166	        if self.mode == "gated":
   167	            gate = torch.sigmoid(self.gate(torch.cat([hidden, projected], dim=-1)))
   168	            return self.normalization(gate * projected + (1.0 - gate) * hidden)
   169	        if self.mode == "attention":
   170	            candidates = torch.stack([hidden, projected], dim=1)
   171	            weights = torch.softmax(self.attention_score(candidates).squeeze(-1), dim=1)
   172	            return self.normalization((weights.unsqueeze(-1) * candidates).sum(dim=1))
   173	        raise ValueError(f"unsupported fusion mode: {self.mode}")

   317	    def _retrieve(
   318	        self,
   319	        *,
   320	        hidden: Tensor,
   321	        memory: ExternalMemoryState,
   322	    ) -> tuple[Tensor, Tensor, ExternalMemoryState]:
   323	        query = self.query_projection(hidden)
   324	        scores = torch.einsum("bd,bsd->bs", query, memory.keys)
   325	        scores = scores / math.sqrt(float(self.key_dim))
   326	        scores = scores.masked_fill(~memory.valid, -1.0e9)
   327	
   328	        top_scores, top_indices = torch.topk(scores, k=self.retrieval_k, dim=1)
   329	        top_valid = torch.gather(memory.valid, dim=1, index=top_indices)
   330	        stable_scores = torch.where(
   331	            top_valid, top_scores, torch.full_like(top_scores, -1.0e9)
   332	        )
   333	        weights = torch.softmax(stable_scores, dim=1) * top_valid.to(scores.dtype)
   334	        weights = weights / weights.sum(dim=1, keepdim=True).clamp_min(1.0e-8)
   335	
   336	        gather_index = top_indices.unsqueeze(-1).expand(-1, -1, self.value_dim)
   337	        top_values = torch.gather(memory.values, dim=1, index=gather_index)
   338	        retrieved = (weights.unsqueeze(-1) * top_values).sum(dim=1)
   339	
   340	        dense_weights = torch.zeros_like(scores)
   341	        dense_weights = dense_weights.scatter_add(1, top_indices, weights)
   342	        memory.retrieval_count = memory.retrieval_count + dense_weights.detach()
   343	        return retrieved, dense_weights, memory

   345	    def _novelty(self, key: Tensor, memory: ExternalMemoryState) -> Tensor:
   346	        normalized_key = F.normalize(key, p=2, dim=-1, eps=1.0e-8)
   347	        normalized_memory = F.normalize(memory.keys, p=2, dim=-1, eps=1.0e-8)
   348	        similarities = torch.einsum("bd,bsd->bs", normalized_key, normalized_memory)
   349	        similarities = similarities.masked_fill(~memory.valid, -1.0)
   350	        maximum_similarity = similarities.max(dim=1).values
   351	        has_memory = memory.valid.any(dim=1)
   352	        novelty = (1.0 - maximum_similarity).clamp(min=0.0, max=2.0)
   353	        return torch.where(has_memory, novelty, torch.ones_like(novelty))
   354	
   355	    def _agreement(self, hidden: Tensor, retrieved: Tensor) -> Tensor:
   356	        projected = self.fusion.memory_projection(retrieved)
   357	        similarity = F.cosine_similarity(hidden, projected, dim=-1, eps=1.0e-8)
   358	        return (similarity + 1.0) * 0.5

   360	    def _write_gate(self, logits: Tensor) -> tuple[Tensor, Tensor, Tensor]:
   361	        probabilities = torch.sigmoid(logits)
   362	        if self.training:
   363	            uniform = torch.rand_like(logits).clamp_(1.0e-6, 1.0 - 1.0e-6)
   364	            logistic_noise = torch.log(uniform) - torch.log1p(-uniform)
   365	            relaxed = torch.sigmoid((logits + logistic_noise) / self.write_temperature)
   366	            hard = (relaxed >= self.write_threshold).to(logits.dtype)
   367	            straight_through = hard.detach() - relaxed.detach() + relaxed
   368	            return probabilities, hard, straight_through
   369	        hard = (probabilities >= self.write_threshold).to(logits.dtype)
   370	        return probabilities, hard, hard

   372	    def _choose_write_slots(
   373	        self,
   374	        *,
   375	        hard_write: Tensor,
   376	        memory: ExternalMemoryState,
   377	        utility_scores: Tensor,
   378	    ) -> tuple[Tensor, Tensor, Tensor]:
   379	        batch_size = memory.batch_size
   380	        allowed = memory.allowed_slot_mask()
   381	        free = allowed & ~memory.valid
   382	        has_free = free.any(dim=1)
   383	        first_free = free.to(torch.int64).argmax(dim=1)
   384	        free_selection = F.one_hot(
   385	            first_free,
   386	            num_classes=memory.max_budget,
   387	        ).to(utility_scores.dtype)
   388	
   389	        eviction_logits = (-utility_scores).masked_fill(~memory.valid, -1.0e9)
   390	        if self.training:
   391	            eviction_selection = F.gumbel_softmax(
   392	                eviction_logits,
   393	                tau=self.write_temperature,
   394	                hard=True,
   395	                dim=1,
   396	            )
   397	        else:
   398	            eviction_slot = utility_scores.masked_fill(
   399	                ~memory.valid,
   400	                torch.inf,
   401	            ).argmin(dim=1)
   402	            eviction_selection = F.one_hot(
   403	                eviction_slot,
   404	                num_classes=memory.max_budget,
   405	            ).to(utility_scores.dtype)
   406	
   407	        selection = torch.where(
   408	            has_free.unsqueeze(1),
   409	            free_selection,
   410	            eviction_selection,
   411	        )
   412	        selection = selection * hard_write.unsqueeze(1)
   413	        selected = selection.argmax(dim=1)
   414	        no_write_value = torch.full_like(selected, -1)
   415	        selected = torch.where(
   416	            hard_write.to(torch.bool),
   417	            selected,
   418	            no_write_value,
   419	        )
   420	        evicted = hard_write.to(torch.bool) & ~has_free
   421	        if selected.shape != (batch_size,):
   422	            raise RuntimeError("internal write-slot selection failure")
   423	        return selected, evicted, selection

   425	    def _apply_writes(
   426	        self,
   427	        *,
   428	        memory: ExternalMemoryState,
   429	        candidate_key: Tensor,
   430	        candidate_value: Tensor,
   431	        write_slots: Tensor,
   432	        selection_weights: Tensor,
   433	        hard_write: Tensor,
   434	        straight_through_gate: Tensor,
   435	        step: int,
   436	    ) -> ExternalMemoryState:
   437	        active = write_slots >= 0
   438	        safe_slots = write_slots.clamp_min(0)
   439	        hard_selected_weights = F.one_hot(
   440	            safe_slots,
   441	            num_classes=memory.max_budget,
   442	        ).to(dtype=memory.keys.dtype)
   443	        hard_selected_weights = hard_selected_weights * active.unsqueeze(1).to(
   444	            hard_selected_weights.dtype
   445	        )
   446	        gate = selection_weights * straight_through_gate.unsqueeze(1)
   447	        gate_3d = gate.unsqueeze(-1)
   448	
   449	        source_key = (
   450	            candidate_key.detach() if self.detach_memory_writes else candidate_key
   451	        )
   452	        source_value = (
   453	            candidate_value.detach() if self.detach_memory_writes else candidate_value
   454	        )
   455	        memory.keys = memory.keys * (1.0 - gate_3d) + source_key.unsqueeze(1) * gate_3d
   456	        memory.values = (
   457	            memory.values * (1.0 - gate_3d) + source_value.unsqueeze(1) * gate_3d
   458	        )
   459	
   460	        hard_selected = hard_selected_weights.to(torch.bool) & hard_write.to(
   461	            torch.bool
   462	        ).unsqueeze(1)
   463	        memory.valid = memory.valid | hard_selected
   464	        memory.utility = memory.utility * (1.0 - hard_selected_weights)
   465	        memory.age = memory.age * (1.0 - hard_selected_weights)
   466	        memory.retrieval_count = memory.retrieval_count * (1.0 - hard_selected_weights)
   467	        memory.last_write_step = torch.where(
   468	            hard_write.to(torch.bool),
   469	            torch.full_like(memory.last_write_step, step),
   470	            memory.last_write_step,
   471	        )
   472	        memory.assert_invariant()
   473	        return memory

   475	    def forward(
   476	        self,
   477	        inputs: Tensor,
   478	        budget: int | Tensor | None = None,
   479	    ) -> BudgetMemROutput:
   480	        """Process ``inputs`` without exposing final task labels to memory control."""
   481	        if inputs.ndim != 3:
   482	            raise ValueError("inputs must have shape [batch, sequence, input_dim]")
   483	        if inputs.shape[-1] != self.input_dim:
   484	            raise ValueError("the last input dimension does not match input_dim")
   485	
   486	        batch_size, sequence_length, _ = inputs.shape
   487	        if sequence_length < 1:
   488	            raise ValueError("sequence length must be positive")
   489	        budgets = self._resolve_budgets(
   490	            budget,
   491	            batch_size=batch_size,
   492	            device=inputs.device,
   493	        )
   494	        normalized_budget = budgets.to(inputs.dtype) / float(self.max_budget)
   495	        budget_embedding = self.budget_conditioner(normalized_budget)
